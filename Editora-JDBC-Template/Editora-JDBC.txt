CREATE SCHEMA `editora-db` ;



CREATE TABLE `editora-db`.`editoras` (
  `ID_EDITORA` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `RAZAO_SOCIAL` VARCHAR(60) NOT NULL COMMENT '',
  `CIDADE` VARCHAR(60) NOT NULL COMMENT '',
  `EMAIL` VARCHAR(60) NOT NULL COMMENT '',
  PRIMARY KEY (`ID_EDITORA`)  COMMENT '',
  UNIQUE INDEX `RAZAO_SOCIAL_UNIQUE` (`RAZAO_SOCIAL` ASC)  COMMENT '',
  UNIQUE INDEX `EMAIL_UNIQUE` (`EMAIL` ASC)  COMMENT '');
