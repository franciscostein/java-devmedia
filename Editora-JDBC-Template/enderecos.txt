CREATE TABLE `editora-db`.`enderecos` (
  `ID_ENDERECO` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `LOGRADOURO` VARCHAR(60) NOT NULL COMMENT '',
  `NUMERO` INT NOT NULL COMMENT '',
  `CIDADE` VARCHAR(60) NOT NULL COMMENT '',
  `ID_AUTOR` INT NOT NULL COMMENT '',
  PRIMARY KEY (`ID_ENDERECO`)  COMMENT '',
  INDEX `FK_ID_AUTOR_idx` (`ID_AUTOR` ASC)  COMMENT '',
  CONSTRAINT `FK_ID_AUTOR`
    FOREIGN KEY (`ID_AUTOR`)
    REFERENCES `editora-db`.`autores` (`ID_AUTOR`)
    ON DELETE CASCADE
    ON UPDATE NO ACTION);
